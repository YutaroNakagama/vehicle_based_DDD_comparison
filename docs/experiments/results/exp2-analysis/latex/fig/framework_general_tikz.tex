\documentclass[tikz,border=3mm]{standalone}
\usepackage[T1]{fontenc}
\usepackage{helvet}
\renewcommand{\familydefault}{\sfdefault}
\usepackage{amsmath,amssymb}
\usepackage{tikz}
\usetikzlibrary{shapes.geometric, positioning, decorations.pathreplacing,
                arrows.meta, calc}

\begin{document}
\begin{tikzpicture}[
  every node/.style={font=\sffamily\small},
  >=Latex,
  db/.style={draw, thick, cylinder, cylinder uses custom fill,
    cylinder body fill=white, shape border rotate=90, aspect=0.08,
    minimum width=3.4cm, minimum height=1.0cm, align=center,
    font=\sffamily\small},
  dbsm/.style={draw, semithick, cylinder, cylinder uses custom fill,
    cylinder body fill=white, shape border rotate=90, aspect=0.08,
    minimum width=1.7cm, minimum height=0.75cm, text width=1.5cm,
    align=center, font=\sffamily\footnotesize},
  pbox/.style={draw, thick,
    align=center, fill=white,
    minimum height=0.8cm,
    inner xsep=8pt, inner ysep=4pt,
    font=\sffamily\small},
  %% Single arrow style: standard ML-journal convention uses one solid
  %% arrow for all data/process flows; the train/val/test distinction
  %% is conveyed by box labels and path geometry, not by line style.
  arr/.style ={-{Latex[length=2mm, width=1.5mm]}, thick},
]

%% ROW 0 — Dataset
\node[db] (dataset) at (0,0)
  {\textbf{Driving Dataset}};

%% ROW 1 — Window segmentation \& labeling
\node[pbox] (label) at (0,-1.3)
  {\textbf{Window Segmentation \& Labeling}};
\draw[arr] (dataset.south) -- (label.north);

%% ROW 2 — Feature extraction
\node[pbox] (feat) at (0,-2.5)
  {\textbf{Feature Extraction}};
\draw[arr] (label.south) -- (feat.north);

%% ROW 3 — Train / Val / Test split (per-subject, chronological)
\node[dbsm] (train) at (-3.0,-4.1) {Train};
\node[dbsm] (val)   at ( 0.0,-4.1) {Val};
\node[dbsm] (test)  at ( 3.0,-4.1) {Test};

%% Fork: trunk down from Feature Extraction, then horizontal split
\coordinate (forkF) at (0,-3.35);
\draw[semithick] (feat.south) -- (forkF);
\draw[semithick] (-3.0,-3.35) -- (3.0,-3.35);
\draw[arr] (-3.0,-3.35) -- (train.north);
\draw[arr] ( 0.0,-3.35) -- (val.north);
\draw[arr] ( 3.0,-3.35) -- (test.north);

%% ROW 4 — Classifier Training & Calibration (train data fits the model;
%% HPO and probability calibration are absorbed into this single block).
\node[pbox] (cls) at (0,-5.7)
  {\textbf{Classifier Training \& Calibration}};
\draw[arr]
  (train.south) -- ++(0,-0.5) -| (cls.north);

%% ROW 5 — Threshold Optimization (val data tunes the decision threshold)
\node[pbox] (thr) at (0,-7.0)
  {\textbf{Threshold Optimization}};
\draw[arr] (cls.south) -- (thr.north);

%% ROW 6 — Evaluation (test data feeds the final evaluation)
\node[pbox] (eval) at (0,-8.2)
  {\textbf{Evaluation}};
\draw[arr] (thr.south) -- (eval.north);

%% ============================================================
%% Auxiliary side branches (val -> threshold; test -> evaluation).
%% Trunks pulled inward so they sit just outside the widest pbox
%% ("Classifier Training & Calibration", edges ~ +/- 3.5).
%%   - Test trunk at y=-4.55 (closest to box row), vertical at x=+5.3
%%   - Val  trunk at y=-4.80 (below test trunk), vertical at x=+4.3
%%   - Train horizontal at y=-4.975 lands on cls.north
%% test_vert (x=+3, y in [-4.55,-4.475]) lies above val_horiz (y=-4.80),
%% so the val and test paths never intersect.
%% ============================================================
%% Val trunk -> Threshold Optimization (sharp 90 degree bends, no rounding)
\draw[semithick]
  (val.south) -- (0.0,-4.80) -- (4.3,-4.80) -- (4.3,-7.0);
\draw[arr] (4.3,-7.0) -- (thr.east);

%% Test trunk -> Evaluation (sharp 90 degree bends, no rounding)
\draw[semithick]
  (test.south) -- (3.0,-4.55) -- (5.3,-4.55) -- (5.3,-8.2);
\draw[arr] (5.3,-8.2) -- (eval.east);

\end{tikzpicture}
\end{document}
